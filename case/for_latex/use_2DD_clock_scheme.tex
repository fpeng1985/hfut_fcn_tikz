\tikzstyle{c}=[rectangle, fill, text=black, minimum size=1cm]
\tikzstyle{c5}=[rectangle, fill, text=black, minimum size=1cm,rotate=-45]
\tikzstyle{c1}=[lightgray!20, c]
\tikzstyle{c2}=[lightgray!50, c] 
\tikzstyle{c3}=[lightgray, c]
\tikzstyle{c4}=[gray, c]
\tikzstyle{clabel}=[rectangle, minimum size=0.2cm, text=black]
\tikzstyle{croute}=[-{Stealth[inset=0pt]}, thick, draw=black!75]
\begin{figure}[!t]%
\subfloat[]{
\begin{tikzpicture}
\tikzmath{
function tdd(\x,\y){
    return (mod(\y, 2)!=0) ? 
           ((mod(\y+1, 4)!=0)?(1+mod(\x+1, 4)):(1+mod(\x+3, 4))) :
           ((mod(\y, 4)==0)?(1+mod(\x, 4)):(1+mod(\x+2, 4)));
};
%
int \i, \j, \c; 
for \i in {0,1,2,3}{
    for \j in {0,1,2,3}{
        \c = tdd(\i,\j);
        {\node[c\c] at(\i,\j){} node(\i\j)[clabel]at(\i,\j){\c};};
    };
};
%
int \rt, \up;
for \i in {0,1,2}{
    for \j in {0,1,2,3}{
        \rt = \i + 1;
        {\draw[croute] (\i\j) -- (\rt\j);};
    };
};
for \i in {0,1,2,3}{
    for \j in {0,1,2}{
        \up = \j + 1;
        {\draw[croute] (\i\j) -- (\i\up);};
    };
};
}
\end{tikzpicture}
}
\subfloat[]{
\begin{tikzpicture}
\tikzmath{
function use(\x,\y){
    return (mod(\y,2)!=0) ? 
           ((mod(\y+1,4)!=0)?(1+mod(\x+1,4)):(1+mod(\x+3,4))) :
           ((mod(\y,4)==0)?(4-mod(\x+3,4)):(4-mod(\x+1,4)));
};
%
int \i, \j, \c; 
for \i in {0,1,2,3}{
    for \j in {0,1,2,3}{
        \c = use(\i,\j);
        {\node[c\c] at(\i,\j){} node(\i\j)[clabel]at(\i,\j){\c};};
    };
};
%
int \rt, \up;
for \i in {0,1,2}{
    for \j in {0,1,2,3}{
        \rt = \i + 1;
        if mod(\j,2) then {
            {\draw[croute] (\i\j) -- (\rt\j);};
        } else {
            {\draw[croute] (\rt\j) -- (\i\j);};
        };
    };
};
for \i in {0,1,2,3}{
    for \j in {0,1,2}{
        \up = \j + 1;
        if mod(\i,2) then {
            {\draw[croute] (\i\up) -- (\i\j);};
        } else {
            {\draw[croute] (\i\j) -- (\i\up);};
        };
    };
};
}
\end{tikzpicture}
}

\end{figure}
